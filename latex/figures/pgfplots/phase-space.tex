\begin{document}
	\begin{tikzpicture}
		\begin{axis}[
			cycle list name=exotic,
		    xlabel={$x=\Re{\alpha}$},
		    ylabel={$p=\Im{\alpha}$},
		    zlabel={$P(x,p)$},
		    width=0.95\linewidth,
		    view={45}{65},
		    grid=major,
		    domain=-1:1,
		    y domain=-1:1,
		]
			\addplot3[
				mesh,

			]{sin(deg(sqrt(x^2+y^2)))/sqrt(x^2+y^2)};
